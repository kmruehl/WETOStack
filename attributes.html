import json

projects = json.load(open("weto_software.json"))

project_name = [project for project, data in projects.items()]
github_account = [data["github_account"] for project, data in projects.items()]
github_repo = [data["github_repo"] for project, data in projects.items()]
active_funding = [data["active_funding"] for project, data in projects.items()]
scope = [data["scope"] for project, data in projects.items()]
primary_use = [data["primary_use"] for project, data in projects.items()]
time_domain = [data["time_domain"] for project, data in projects.items()]

data = {
    "project_name": project_name,
    "github_account": github_account,
    "github_repo": github_repo,
    "primary_use": primary_use,
    "time_domain": time_domain,
    "active_funding": active_funding,
    "scope": scope,
}
source = ColumnDataSource(data=data)
columns = [
    TableColumn(field="project_name", title="Name"),
    TableColumn(field="github_account", title="GH Account"),
    TableColumn(field="github_repo", title="GH Repo"),
    TableColumn(field="primary_use", title="Primary Use"),
    TableColumn(field="time_domain", title="Time Domain"),
    TableColumn(field="active_funding", title="Active Funding"),
    TableColumn(field="scope", title="Scope"),
    # TableColumn(field="dates", title="Date", formatter=DateFormatter()),
]
data_table = DataTable(source=source, columns=columns) #, width=400, height=280)

show(data_table)
